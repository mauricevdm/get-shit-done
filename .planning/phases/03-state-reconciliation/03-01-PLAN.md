    ```javascript
    #!/usr/bin/env node

    const { unified } = require('unified');
    const remarkParse = require('remark-parse');
    const remarkStringify = require('remark-stringify');
    const remarkGfm = require('remark-gfm');
    const { headingRange } = require('mdast-util-heading-range');

    // Create processor with GFM support
    const processor = unified()
      .use(remarkParse)
      .use(remarkGfm);

    const stringifier = unified()
      .use(remarkParse)
      .use(remarkGfm)
      .use(remarkStringify, {
        bullet: '-',
        listItemIndent: 'one'
      });

    /**
     * Parse STATE.md content into mdast tree
     */
    function parseStateFile(content) {
      return processor.parse(content);
    }

    /**
     * Extract section content by heading text
     * Returns array of nodes between heading and next same-level heading, or null if not found
     */
    function extractSection(tree, headingText) {
      let sectionNodes = null;

      headingRange(tree, headingText, (start, nodes, end) => {
        sectionNodes = {
          heading: start,
          content: nodes,
          end: end
        };
      });

      return sectionNodes;
    }

    /**
     * Serialize section nodes back to markdown
     */
    function serializeSection(section) {
      if (!section || !section.content) return '';

      // Create temporary root with just the section content
      const tempTree = {
        type: 'root',
        children: section.content
      };

      return stringifier.stringify(tempTree);
    }

    module.exports = { parseStateFile, extractSection, serializeSection };
    ```
